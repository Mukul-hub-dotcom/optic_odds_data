CREATE TABLE vi_ex_master_sports (
  sports_id INT AUTO_INCREMENT PRIMARY KEY,
  sport_guid INT DEFAULT NULL,
  bet365_sport_guid INT DEFAULT 0,
  lsport_guid INT NOT NULL,
  sports_name VARCHAR(100) NOT NULL,
  updated_date TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  active TINYINT(1) NOT NULL DEFAULT 0,
  `order` INT NOT NULL DEFAULT 0,
  provider_name VARCHAR(55) DEFAULT NULL,
  sportbook_betting TINYINT(1) NOT NULL DEFAULT 1,
  exchange_betting TINYINT(1) NOT NULL DEFAULT 1,
  casino_betting TINYINT(1) NOT NULL DEFAULT 1,
  fancy_betting TINYINT(1) NOT NULL DEFAULT 0,
  min_odd_limit FLOAT NOT NULL DEFAULT 0,
  max_odd_limit FLOAT NOT NULL DEFAULT 0,
  agent_min_odd_limit FLOAT NOT NULL DEFAULT 0,
  agent_max_odd_limit FLOAT NOT NULL DEFAULT 0,
  icon VARCHAR(50) DEFAULT NULL
);
