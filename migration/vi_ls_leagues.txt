CREATE TABLE vi_ls_leagues (
  league_id SERIAL PRIMARY KEY,
  league_uid INT DEFAULT NULL,
  location_id INT NOT NULL,
  sport_guid INT NOT NULL,
  league_abbr VARCHAR(100) NOT NULL,
  league_name VARCHAR(255) DEFAULT NULL,
  region VARCHAR(55) DEFAULT NULL,
  active BOOLEAN NOT NULL DEFAULT FALSE,
  agent_active BOOLEAN NOT NULL DEFAULT TRUE,
  updated_date TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  is_deleted BOOLEAN NOT NULL DEFAULT FALSE,
  is_agent_deleted BOOLEAN NOT NULL DEFAULT FALSE
);
