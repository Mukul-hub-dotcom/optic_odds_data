CREATE TABLE vi_ls_leagues (
  league_id INT AUTO_INCREMENT PRIMARY KEY,
  league_uid VARCHAR(100) UNIQUE,
  location_id INT NOT NULL,
  sport_guid INT NOT NULL,
  league_abbr VARCHAR(100) NOT NULL,
  league_name VARCHAR(255) DEFAULT NULL,
  region VARCHAR(55) DEFAULT NULL,
  active TINYINT(1) NOT NULL DEFAULT 0,
  agent_active TINYINT(1) NOT NULL DEFAULT 1,
  updated_date TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  is_deleted TINYINT(1) NOT NULL DEFAULT 0,
  is_agent_deleted TINYINT(1) NOT NULL DEFAULT 0
);
