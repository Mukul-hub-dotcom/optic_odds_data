CREATE TABLE vi_lsport_market_odds (
    odd_id SERIAL PRIMARY KEY,
    event_id VARCHAR(255),
    market_id VARCHAR(255),
    market_name VARCHAR(255),
    market_key VARCHAR(255),
    line NUMERIC DEFAULT 0.00,
    id VARCHAR(255) UNIQUE,
    provider_id INTEGER DEFAULT 1,
    status INTEGER DEFAULT 0,
    name VARCHAR(255),
    market_line VARCHAR(255) DEFAULT NULL,
    odds NUMERIC DEFAULT 0,
    sort_priority INTEGER DEFAULT NULL,
    header VARCHAR(255) DEFAULT NULL,
    handicap NUMERIC DEFAULT NULL,
    is_suspended INTEGER DEFAULT 0,
    fraction_odds VARCHAR(255) DEFAULT NULL,
    settlement_status INTEGER DEFAULT 0,
    ls_settlement INTEGER DEFAULT 1,
    settlement_result VARCHAR(255) DEFAULT NULL,
    admin_status INTEGER DEFAULT 1,
    agent_status INTEGER DEFAULT 1,
    lock_betting INTEGER DEFAULT 0,
    agent_lock_betting INTEGER DEFAULT 0,
    settlement_by VARCHAR(255) DEFAULT NULL,
    settlement_date TIMESTAMP DEFAULT NULL,
    rolledback_by VARCHAR(255) DEFAULT NULL,
    rolledback_date TIMESTAMP DEFAULT NULL
);
