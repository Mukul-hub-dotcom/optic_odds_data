CREATE TABLE vi_lsport_market_odds (
    odd_id INT AUTO_INCREMENT PRIMARY KEY,
    event_id VARCHAR(255),
    market_id VARCHAR(255),
    market_name VARCHAR(255),
    market_key VARCHAR(255),
    line DECIMAL(10, 2) DEFAULT 0.00,
    id VARCHAR(255) UNIQUE,
    provider_id INT DEFAULT 1,
    status INT DEFAULT 0,
    name VARCHAR(255),
    market_line VARCHAR(255) DEFAULT NULL,
    odds DECIMAL(10, 2) DEFAULT 0,
    sort_priority INT DEFAULT NULL,
    header VARCHAR(255) DEFAULT NULL,
    handicap DECIMAL(10, 2) DEFAULT NULL,
    is_suspended INT DEFAULT 0,
    fraction_odds VARCHAR(255) DEFAULT NULL,
    settlement_status INT DEFAULT 0,
    ls_settlement INT DEFAULT 1,
    settlement_result VARCHAR(255) DEFAULT NULL,
    admin_status INT DEFAULT 1,
    agent_status INT DEFAULT 1,
    lock_betting INT DEFAULT 0,
    agent_lock_betting INT DEFAULT 0,
    settlement_by VARCHAR(255) DEFAULT NULL,
    settlement_date TIMESTAMP NULL DEFAULT NULL,
    rolledback_by VARCHAR(255) DEFAULT NULL,
    rolledback_date TIMESTAMP NULL DEFAULT NULL
);
